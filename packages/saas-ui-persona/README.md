# @saas-ui/persona

A Persona component for Chakra UI

## Installation

```sh
$ yarn add @saas-ui/persona

#or

$ npm i @saas-ui/persona --save
```

## Usage

```tsx
import { Persona } from '@saas-ui/persona'

export default const App = () => {
  return (
    <Box>
      <Persona
        name="Eelco Wiersma"
        secondaryLabel="Founder"
        tertiaryLabel="On a roadtrip"
        presence="offline"
        isOutOfOffice
        />
    </Box>
  )
}
```
