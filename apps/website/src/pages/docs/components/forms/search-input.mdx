## Import

```ts
import { SearchInput } from '@saas-ui/react'
```

## Usage

### Basic

```jsx
<SearchInput placeholder="Search" />
```

### Controlled

```jsx
function Search() {
  const [value, setValue] = React.useState('')

  return (
    <SearchInput
      placeholder="Search"
      value={value}
      onChange={(e) => setValue(e.target.value)}
      onReset={() => setValue('')}
    />
  )
}
```
