## Import

```ts
import { Web3Address } from '@saas-ui/web3'
```

## Usage

```jsx
<Web3Address address="0x71C7656EC7ab88b098defB751B7401B5f6d8976F" />
```

### Custom length

```jsx
<Web3Address
  address="0x71C7656EC7ab88b098defB751B7401B5f6d8976F"
  startLength={6}
  endLength={3}
/>
```

### In a button

```jsx
<Button>
  <Web3Address address="0x71C7656EC7ab88b098defB751B7401B5f6d8976F" />
</Button>
```
